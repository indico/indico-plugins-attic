{% trans %}General Syntax{% endtrans %}

  • FIELD:TERM - {% trans %}field contains term{% endtrans %}
  • FIELD=TERM - {% trans %}field equals term{% endtrans %}
  • -FIELD:TERM - {% trans %}field does not contain term{% endtrans %}
  • FIELD<>TERM - {% trans %}field does not equal term{% endtrans %}

{% trans %}A term containing spaces should be enclosed in quotes.{% endtrans %} {% trans %}Multiple criteria can be separated using AND (default, can be omitted) or OR.{% endtrans %}

{% trans %}Examples{% endtrans %}

  • Location:123-2-025
  • Author:"John Doe" Title:Indico
  • -Author:"Bobby Boring" Title:Indico
  • Author:Doe Format:PDF
